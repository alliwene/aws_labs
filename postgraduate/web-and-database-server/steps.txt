- Create key pair 
	aws ec2 create-key-pair --key-name mykey --query 'KeyMaterial' --output text > mykey.pem

- Create stack
	aws cloudformation create-stack \
	--stack-name db-stack \
	--template-body file://formation.yml \
	--disable-rollback --profile sandbox

	--on-failure DO_NOTHING 

- Track create progress 
	watch -n 5 -d \
	aws cloudformation describe-stack-resources \
	--stack-name db-stack \
	--query 'StackResources[*].[ResourceType,ResourceStatus]' \
	--output table --profile sandbox

- Check resources that failed to create
	aws cloudformation describe-stack-events \
	--stack-name db-stack \
	--query "StackEvents[?ResourceStatus == 'CREATE_FAILED']" --profile sandbox

- Update stack
	aws cloudformation update-stack --stack-name db-stack --template-body file://formation.yml --disable-rollback --profile sandbox

- Delete stack 
	aws cloudformation delete-stack --stack-name db-stack --profile sandbox

- Get rds endpoint 
	aws rds describe-db-instances --db-instance-identifier tutorial-db-instance --query 'DBInstances[0].Endpoint.Address' --profile sandbox
